            public static function getImage(target:UIComponent):XImage
            {
                while (!(target is XImage) && target != null && target.parent != null)
                {
                    target = target.parent as UIComponent;
                }
                return target as XImage;
            }
            
            
            override public function set model(value:IModel):void
            {
                super.model = value;
                _imageModel = value as XImageModel;
                _bounds = _imageModel.bounds;
                _constraints = _imageModel.constraints;
                _bevel = _imageModel.fx.bevel;
                _blur = _imageModel.fx.blur;
                _dropShadow = _imageModel.fx.dropShadow;
                _glow = _imageModel.fx.glow;
            }
            
            
            public function get isEmbedded():Boolean

            {
                return _constraints;
            }
